function gl_form_send() {

    var stext = gl_form.S_Text.value;
    for (var i = 0; i &lt; stext.length; i++) {
        if (stext.indexOf("+") != -1) {
            stext = stext.replace("+", "%2B")
        }
        if (stext.indexOf("#") != -1) {
            stext = stext.replace("#", "%23")
        }

    }

    stext = stext.replace("(주)", "");
    stext = stext.replace("㈜", "");
    if ((stext.length == 1) &amp;&amp; (stext == "'")) {
        alert("\n 키워드를 입력해 주세요.!");
        gl_form.S_Text.value = "";
        gl_form.S_Text.focus();
        return false;
    } else {
        gl_form.S_Text.value = stext;
    }

    }

    gl_form.submit();

}
